## 1. Conditional


```go
a := 10

if a > 20 {
    fmt.Println(">")
} else if a < 20 {
    fmt.Println("<")
} else {
    fmt.Println("=")
}
```


## 2. Statements in if

```go 
x := "hello go!"

if count := len(x); count > 0 {
    fmt.Println("Yes")
}
```

---

```go
if s, ok := v.(string); ok {
    fmt.Println("строка:", s)
}
```

--- 

```go
if _, err := doThing(); err != nil {
    fmt.Println("Uh oh")
}
```


## 3. Switch

```go
x := 42.0
switch x {
case 0:
case 1, 2:
    fmt.Println("Multiple matches")
case 42:   // Don't "fall through".
    fmt.Println("reached")
case 43:
    fmt.Println("Unreached")
    
default:
    fmt.Println("Optional")
}
```

> See: [Switch](https://go.dev/wiki/Switch)


## 4. For loop

```go
for i := 0; i <= 10; i++ {
  fmt.Println("i: ", i)
}
```

## 5. For-Range loop

```go
nums := []int{2, 3, 4}
sum := 0
for _, num := range nums {
    sum += num
}
fmt.Println("sum:", sum)
```


## 6. While loop

```go
i := 1
for i <= 3 {
    fmt.Println(i)
    i++
}
```

---
```go
func main() {
	mylist := list.New()
	mylist.PushBack(1)
	mylist.PushBack(2)
	mylist.PushBack(3)
	for temp := mylist.Front(); temp != nil; temp = temp.Next() {
		fmt.Println(temp.Value) 
	}
}
```

## 7. Continue keyword

```go
for i := 0; i <= 5; i++ {
    if i % 2 == 0 {
        continue
    }
    fmt.Println(i)
}
```


## 8. Break keyword


```go
for {
    fmt.Println("loop")
    break
}
```
